### 最小実装 — タロット占い

```python
import os
from anthropic import Anthropic
from dotenv import load_dotenv

load_dotenv()

client = Anthropic()

def read_tarot(situation: str, theme: str) -> str:
    """
    シンプルなタロット占い

    Args:
        situation: 相談者の状況（例: "転職を検討している30代会社員"）
        theme: 最も知りたいこと（例: "転職の時期と方向性"）

    Returns:
        タロットリーディング結果テキスト
    """
    prompt = f"""あなたはプロのタロットリーダーです。
78枚のウェイト＝スミス版タロットから、ケルト十字スプレッドで10枚を引いてください。

【相談者情報】
状況: {situation}
テーマ: {theme}

以下の形式で出力してください：

## 引いたカード（10枚）
各ポジション番号と意味、カード名、正/逆位置を表形式で示す

## 各カードの解釈（各100字以上）
ポジションごとに詳細な解釈

## 統合メッセージ
全体のストーリーと今後3ヶ月の流れ

## アドバイス
具体的な行動指針を3点

カードはランダムに選択し、正逆もランダムに決定してください。"""

    message = client.messages.create(
        model="claude-opus-4-5",
        max_tokens=2000,
        messages=[
            {"role": "user", "content": prompt}
        ]
    )

    return message.content[0].text


# 実行例
if __name__ == "__main__":
    result = read_tarot(
        situation="転職を検討している30代会社員。現職は安定しているが成長を感じられない",
        theme="転職の最適なタイミングと進むべき方向性"
    )
    print(result)
```

## 第2章：パイプライン型実装 — 数秘術

数秘術は計算ロジックとAI解釈を分離できるため、パイプライン設計の練習に最適。

```python
from datetime import date
from anthropic import Anthropic

client = Anthropic()


def calculate_life_path_number(birth_date: date) -> int:
    """
    ライフパスナンバーを計算する

    数秘術の基本: 生年月日の全数字を1桁になるまで足す
    ただし 11, 22, 33 はマスターナンバーとして保持
    """
    date_str = birth_date.strftime("%Y%m%d")
    total = sum(int(d) for d in date_str)

    # マスターナンバーチェック
    while total > 9 and total not in (11, 22, 33):
        total = sum(int(d) for d in str(total))

    return total


def calculate_destiny_number(full_name: str) -> int:
    """
    デスティニーナンバーを計算する（ピタゴリアン数秘術）
    """
    pythagorean = {
        'a': 1, 'b': 2, 'c': 3, 'd': 4, 'e': 5,
        'f': 6, 'g': 7, 'h': 8, 'i': 9, 'j': 1,
        'k': 2, 'l': 3, 'm': 4, 'n': 5, 'o': 6,
        'p': 7, 'q': 8, 'r': 9, 's': 1, 't': 2,
        'u': 3, 'v': 4, 'w': 5, 'x': 6, 'y': 7, 'z': 8
    }

    total = sum(
        pythagorean.get(c.lower(), 0)
        for c in full_name
        if c.isalpha()
    )

    while total > 9 and total not in (11, 22, 33):
        total = sum(int(d) for d in str(total))

    return total


def interpret_numerology(
    birth_date: date,
    name: str,
    question: str
) -> str:
    """
    数秘術パイプライン: 計算 → AI解釈
    """
    life_path = calculate_life_path_number(birth_date)
    destiny = calculate_destiny_number(name)

    # 現在の年の個人年数計算
    current_year = date.today().year
    personal_year_raw = (
        birth_date.day + birth_date.month + sum(int(d) for d in str(current_year))
    )
    personal_year = personal_year_raw
    while personal_year > 9 and personal_year not in (11, 22, 33):
        personal_year = sum(int(d) for d in str(personal_year))

    prompt = f"""あなたはピタゴリアン数秘術の専門家です。
以下の数値データをもとに、詳細なリーディングを行ってください。

【数値データ】
- 生年月日: {birth_date.strftime('%Y年%m月%d日')}
- ライフパスナンバー: {life_path}
- デスティニーナンバー: {destiny}
- 個人年数（{current_year}年）: {personal_year}
- 相談者名（ローマ字）: {name}

【相談内容】
{question}

以下を含む詳細な解釈を提供してください：
1. ライフパスナンバー{life_path}の意味と人生の使命
2. デスティニーナンバー{destiny}が示す才能と課題
3. 個人年数{personal_year}が示す{current_year}年のテーマ
4. 数値の組み合わせから見えるパターン
5. 相談内容へのアドバイス（具体的かつ実行可能な3点）"""

    message = client.messages.create(
        model="claude-sonnet-4-5",
        max_tokens=1500,
        messages=[{"role": "user", "content": prompt}]
    )

    return {
        "life_path": life_path,
        "destiny": destiny,
        "personal_year": personal_year,
        "interpretation": message.content[0].text
    }


# 実行例
if __name__ == "__main__":
    result = interpret_numerology(
        birth_date=date(1990, 5, 15),
        name="Tanaka Hanako",
        question="副業を始めるタイミングと、向いている分野を知りたい"
    )
    print(f"ライフパスナンバー: {result['life_path']}")
    print(f"個人年数: {result['personal_year']}")
    print("\n--- 解釈 ---")
    print(result['interpretation'])
```

## 第3章：コンテンツ量産パターン

占いコンテンツを自動化するには、1つのテンプレートから複数バリエーションを生成する設計が効率的。

```python
import json
from anthropic import Anthropic

client = Anthropic()


def generate_fortune_batch(
    fortune_type: str,
    zodiac_signs: list[str],
    theme: str,
    output_format: str = "markdown"
) -> dict[str, str]:
    """
    複数の星座/干支/タイプ向けに占いコンテンツを一括生成

    Args:
        fortune_type: 占いの種類（"tarot", "numerology", "astrology"等）
        zodiac_signs: 生成対象のリスト
        theme: コンテンツのテーマ
        output_format: "markdown" または "json"

    Returns:
        {星座名: コンテンツ} の辞書
    """
    results = {}

    for sign in zodiac_signs:
        prompt = f"""あなたはプロの占い師です。
{sign}の方向けに、{theme}に関する{fortune_type}占いを書いてください。

条件:
- 500〜800字
- 具体的なアドバイスを含む
- 前向きかつ現実的なトーン
- 「{sign}の方へ」で始める
- 末尾に「今週のラッキーアクション」1点を追記

出力はMarkdownで。"""

        message = client.messages.create(
            model="claude-haiku-4-5",  # 量産はHaikuでコスト削減
            max_tokens=800,
            messages=[{"role": "user", "content": prompt}]
        )

        results[sign] = message.content[0].text
        print(f"✓ {sign} 完了")

    return results


# 実行例: 12星座の週間運勢を一括生成
ZODIAC_SIGNS = [
    "牡羊座", "牡牛座", "双子座", "蟹座",
    "獅子座", "乙女座", "天秤座", "蠍座",
    "射手座", "山羊座", "水瓶座", "魚座"
]

if __name__ == "__main__":
    fortunes = generate_fortune_batch(
        fortune_type="タロット",
        zodiac_signs=ZODIAC_SIGNS[:3],  # テスト用に3星座
        theme="2026年4月の仕事運と金運"
    )

    # ファイルに保存
    with open("fortune_output.json", "w", encoding="utf-8") as f:
        json.dump(fortunes, f, ensure_ascii=False, indent=2)

    print(f"\n{len(fortunes)}件の占いコンテンツを生成しました")
```

## 第4章：よくあるトラブルと対策

### エラー1: `anthropic.APIStatusError: 529`

**原因**: APIの過負荷（稀）
**対策**: 指数バックオフで再試行

```python
import time
from anthropic import APIStatusError

def call_with_retry(client, prompt: str, max_retries: int = 3) -> str:
    for attempt in range(max_retries):
        try:
            message = client.messages.create(
                model="claude-haiku-4-5",
                max_tokens=800,
                messages=[{"role": "user", "content": prompt}]
            )
            return message.content[0].text
        except APIStatusError as e:
            if attempt == max_retries - 1:
                raise
            wait_time = 2 ** attempt  # 1秒, 2秒, 4秒
            print(f"リトライ {attempt + 1}/{max_retries} — {wait_time}秒後")
            time.sleep(wait_time)
```

### エラー2: 出力が毎回同じパターンになる

**原因**: プロンプトが固定的すぎる
**対策**: ランダム要素を明示的に指示する

```python
import random

def add_variation(prompt: str) -> str:
    """プロンプトにバリエーション要素を追加"""
    variations = [
        "今回は特に感情面に注目して解釈してください",
        "キャリアと人間関係の両面から読み解いてください",
        "中長期（6ヶ月〜1年）の視点で解釈してください",
        "具体的な行動ステップに重点を置いてください",
    ]
    variation = random.choice(variations)
    return f"{prompt}\n\n補足: {variation}"
```

### エラー3: コスト管理

**対策**: トークン数の事前推定と上限設定

```python
def estimate_cost(
    input_tokens: int,
    output_tokens: int,
    model: str = "claude-haiku-4-5"
) -> float:
    """APIコストをUSDで概算"""
    pricing = {
        "claude-haiku-4-5": {"input": 0.0008, "output": 0.004},
        "claude-sonnet-4-5": {"input": 0.003, "output": 0.015},
        "claude-opus-4-5": {"input": 0.015, "output": 0.075},
    }

    rates = pricing.get(model, pricing["claude-haiku-4-5"])
    cost = (
        (input_tokens / 1000) * rates["input"] +
        (output_tokens / 1000) * rates["output"]
    )
    return cost

# 月100件生成の場合
monthly_cost = estimate_cost(500, 800, "claude-haiku-4-5") * 100
print(f"月間コスト概算: ${monthly_cost:.2f} (約¥{monthly_cost * 150:.0f})")
# → 月間コスト概算: $0.44 (約¥66)
```
